exact {
  let x00 = flip 1.0 / 2.0 in
  let x01_10 = sample {
      x00 <- exact { x00 };
      p01 <- if  x00 then 1.0 / 3.0 else 1.0 / 4.0;
      p10 <- if !x00 then 1.0 / 5.0 else 1.0 / 6.0;
      x01 ~ bern(p01);
      x10 ~ bern(p10);
      (x01, x10)
  } in
  let x01 = x01_10[0] in
  let x10 = x01_10[1] in
  let x11 =
      if  x10  &&  x01 then flip 3.0 / 7.0 else
      if  x10  && !x01 then flip 3.0 / 8.0 else
      if !x10  &&  x01 then flip 3.0 / 9.0 else
                            flip 3.0 / 11.0 in
  (x00, x01, x10, x11)
